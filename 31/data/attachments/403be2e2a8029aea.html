    var csrfUrl = '/auth/regenerateLoginCsrfToken';
    localStorage.clear();

    function calculateUserTimeZoneOffset() {
        var myDate = new Date();
        var offset = (-1) * myDate.getTimezoneOffset() / 60;
        return offset;
    }
            
    function addHint(inputObject, hintImageURL) {
        if (inputObject.val() == '') {
            inputObject.css('background', "url('" + hintImageURL + "') no-repeat 10px 3px");
        }
    }
            
    function removeHint() {
       $('.form-hint').css('display', 'none');
    }
    
    function showMessage(message) {
        if ($('#spanMessage').size() == 0) {
            $('&lt;span id="spanMessage"&gt;&lt;/span&gt;').insertAfter('#btnLogin');
        }

        $('#spanMessage').html(message);
    }
    
    function validateLogin() {
        var isEmptyPasswordAllowed = false;
        
        if ($('#txtUsername').val() == '') {
            showMessage('Username cannot be empty');
            return false;
        }
        if ($( "#txtPassword" ).length) {
            var txtPasswordExist = true;
        } else {
            var txtPasswordExist = false;
        }
    
        if (!isEmptyPasswordAllowed) {
            if ($('#txtPassword').val() == '' || !txtPasswordExist) {
                showMessage('Password cannot be empty');
                return false;
            }
        }
        return true;
    }

    function submitLoginForm(event, csrfTokenUpdated) {

        var formValid = validateLogin();
        if (formValid) {

            // Continue submit if CSRF token is updated
            if (csrfTokenUpdated) {
                return true;
            }

            // If csrf is invalid, make AJAX call to fetch csrf token and abort this submit by returning false
            if (!getCsrfToken()) {
                $.post(csrfUrl, function(token) {
                    $('#login__csrf_token').val(token);

                    $('#frmLogin').trigger('submit', [true]);
                });
                return false;
            }

            removeCSRFToken();
            return true;
        }
        return false;
    }

    function getCsrfToken() {
        var csrfToken = localStorage.getItem('login_csrf');

        try {
            var now = new Date();
            var csrfTokenObj = JSON.parse(csrfToken);

            if (csrfTokenObj &amp;&amp; csrfTokenObj.expire &amp;&amp; csrfTokenObj.expire &gt; now.getTime()) {
                $('#login__csrf_token').val(csrfTokenObj.token);
                return true;
            }

        } catch (e) {
            removeCSRFToken();
        }

        return false;
    }

    function removeCSRFToken() {
        localStorage.removeItem('login_csrf');
    }

    function saveCSRFToken(token) {
        var now = new Date();
        var expireInMS = now.getTime() + 2 * 60 * 1000;

        localStorage.setItem('login_csrf', JSON.stringify({token: token, expire: expireInMS}));
    }
    
    $(document).ready(function() {

        /* Fading out main messages */
        jQuery(document).on({

            }

        /* Set cookie with csrf token to expire in 2 minutes */
        saveCSRFToken($('#login__csrf_token').val());

        /*Set a delay to compatible with chrome browser*/

        $('#txtUsername').focus(function() {
            removeHint();

             removeHint();
        });
        
        $('.form-hint').click(function(){
            removeHint();
            $('#txtUsername').focus();
        });
        
        $('#hdnUserTimeZoneOffset').val(calculateUserTimeZoneOffset().toString());
        
        $('#frmLogin').submit(submitLoginForm);
    });

    function checkSavedUsernames(){
        if ($('#txtUsername').val() != '') {
            removeHint();
        }
    }
